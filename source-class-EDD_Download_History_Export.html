 22 class EDD_Download_History_Export extends EDD_Export {
 23     /**
 24      * Our export type. Used for export-type specific filters/actions
 25      *
 26      * @var string
 27      * @since 1.4.4

 32     /**
 33      * Set the export headers
 34      *
 35      * @access public
 36      * @since 1.4.4
 37      * @return void
 38      */
 39     public function headers() {
 40         ignore_user_abort( true );
 41 
 42         if ( ! edd_is_func_disabled( 'set_time_limit' ) && ! ini_get( 'safe_mode' ) )
 43             set_time_limit( 0 );
 44 
 45         $month = isset( $_POST['month'] ) ? absint( $_POST['month'] ) : date( 'n' );
 46         $year  = isset( $_POST['year']  ) ? absint( $_POST['year']  ) : date( 'Y' );
 47 
 48         nocache_headers();
 49         header( 'Content-Type: text/csv; charset=utf-8' );
 50         header( 'Content-Disposition: attachment; filename=edd-export-' . $this->export_type . '-' . $month . '-' . $year . '.csv' );
 51         header( "Expires: 0" );
 52     }
 53 
 54 
 55     /**
 56      * Set the CSV columns
 57      *
 58      * @access public
 59      * @since 1.4.4
 60      * @return array $cols All the columns
 61      */
 62     public function csv_cols() {
 63         $cols = array(
 64             'date'     => __( 'Date',   'edd' ),
 65             'user'     => __( 'Downloaded by', 'edd' ),
 66             'ip'       => __( 'IP Address', 'edd' ),
 67             'download' => __( 'Product', 'edd' ),
 68             'file'     => __( 'File', 'edd' )
 69         );
 70         return $cols;
 71     }
 72 
 73     /**
 74      * Get the Export Data
 75      *
 76      * @access public
 77      * @since 1.4.4
 78      * @global object $edd_logs EDD Logs Object
 79      * @return array $data The data for the CSV file
 80      */
 81     public function get_data() {
 82         global $edd_logs;
 83 
 84         $data = array();
 85 
 86         $args = array(
 87             'nopaging' => true,
 88             'log_type' => 'file_download',
 89             'monthnum' => isset( $_POST['month'] ) ? absint( $_POST['month'] ) : date( 'n' ),
 90             'year'     => isset( $_POST['year'] ) ? absint( $_POST['year'] ) : date( 'Y' )
 91         );
 92 
 93         $logs = $edd_logs->get_connected_logs( $args );
 94 
 95         if ( $logs ) {
 96             foreach ( $logs as $log ) {
 97                 $user_info = get_post_meta( $log->ID, '_edd_log_user_info', true );
 98                 $files     = edd_get_download_files( $log->post_parent );
 99                 $file_id   = (int) get_post_meta( $log->ID, '_edd_log_file_id', true );
100                 $file_name = isset( $files[ $file_id ]['name'] ) ? $files[ $file_id ]['name'] : null;
101                 $user      = get_userdata( $user_info['id'] );
102                 $user      = $user ? $user->user_login : $user_info['email'];
103 
104                 $data[]    = array(
105                     'date'     => $log->post_date,
106                     'user'     => $user,
107                     'ip'       => get_post_meta( $log->ID, '_edd_log_ip', true ),
108                     'download' => get_the_title( $log->post_parent ),
109                     'file'     => $file_name
110                 );
111             }
112         }
113 
114         $data = apply_filters( 'edd_export_get_data', $data );
115         $data = apply_filters( 'edd_export_get_data_' . $this->export_type, $data );
116 
117         return $data;
118     }

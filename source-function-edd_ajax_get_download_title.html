 30 /**
 31  * Get AJAX URL
 32  *
 33  * @since 1.3
 34  * @return string
 35 */
 36 function edd_get_ajax_url() {
 37     $scheme = defined( 'FORCE_SSL_ADMIN' ) && FORCE_SSL_ADMIN ? 'https' : 'admin';
 38 
 39     $current_url = edd_get_current_page_url();
 40     $ajax_url    = admin_url( 'admin-ajax.php', $scheme );
 41 
 42     if ( preg_match( '/^https/', $current_url ) && ! preg_match( '/^https/', $ajax_url ) ) {
 43         $ajax_url = preg_replace( '/^http/', 'https', $ajax_url );
 44     }
 45 
 46     return apply_filters( 'edd_ajax_url', $ajax_url );
 47 }

 65 /**
 66  * Adds item to the cart via AJAX.
 67  *
 68  * @since 1.0
 69  * @return void
 70  */
 71 function edd_ajax_add_to_cart() {
 72     if ( isset( $_POST['download_id'] ) && check_ajax_referer( 'edd_ajax_nonce', 'nonce' ) ) {
 73         global $post;
 74 
 75         $to_add = array();
 76 
 77         if ( isset( $_POST['price_ids'] ) && is_array( $_POST['price_ids'] ) ) {
 78             foreach ( $_POST['price_ids'] as $price ) {
 79                 $to_add[] = array( 'price_id' => $price );
 80             }
 81         }
 82 
 83         foreach ( $to_add as $options ) {
 84             if ( ! edd_item_in_cart( $_POST['download_id'], $options ) ) {
 85 
 86                 if( $_POST['download_id'] == $options['price_id'] )
 87                     $options = array();
 88 
 89                 $key          = edd_add_to_cart( $_POST['download_id'], $options );
 90 
 91                 $item         = array(
 92                     'id'      => $_POST['download_id'],
 93                     'options' => $options
 94                 );
 95 
 96                 $item = apply_filters( 'edd_ajax_pre_cart_item_template', $item );
 97 
 98                 $cart_item    = edd_get_cart_item_template( $key, $item, true );
 99 
100                 echo $cart_item;
101             } else {
102                 echo 'incart';
103             }
104         }
105     }
106     edd_die();
107 }
108 add_action( 'wp_ajax_edd_add_to_cart', 'edd_ajax_add_to_cart' );

111 /**
112  * Validates the supplied discount sent via AJAX.
113  *
114  * @since 1.0
115  * @return void
116  */
117 function edd_ajax_apply_discount() {
118     if ( isset( $_POST['code'] ) && check_ajax_referer( 'edd_checkout_nonce', 'nonce' ) ) {
119         $user = isset( $_POST['user'] ) ? $_POST['user'] : $_POST['email'];
120 
121         $return = array(
122             'msg'  => '',
123             'code' => $_POST['code']
124         );
125 
126         if ( edd_is_discount_used( $_POST['code'], $user ) ) {  // Called twice if discount is not used (again by edd_is_discount_valid) but allows for beter usr msg and less execution if discount is used.
127             $return['msg']  = __('This discount code has been used already', 'edd');
128         } else {
129             if ( edd_is_discount_valid( $_POST['code'], $user ) ) {
130                 $discount  = edd_get_discount_by_code( $_POST['code'] );
131                 $amount    = edd_format_discount_rate( edd_get_discount_type( $discount->ID ), edd_get_discount_amount( $discount->ID ) );
132                 $discounts = edd_set_cart_discount( $_POST['code'] );
133                 $total     = edd_get_cart_total( $discounts );
134 
135                 $return = array(
136                     'msg'    => 'valid',
137                     'amount' => $amount,
138                     'total'  => html_entity_decode( edd_currency_filter( edd_format_amount( $total ) ), ENT_COMPAT, 'UTF-8' ),
139                     'code'   => $_POST['code'],
140                     'html'   => edd_get_cart_discounts_html( $discounts )
141                 );
142             } else {
143                 $return['msg']  = __('The discount you entered is invalid', 'edd');
144             }
145         }
146         echo json_encode($return);
147     }
148     edd_die();
149 }
150 add_action( 'wp_ajax_edd_apply_discount', 'edd_ajax_apply_discount' );
151 add_action( 'wp_ajax_nopriv_edd_apply_discount', 'edd_ajax_apply_discount' );

255 /**
256  * Check for Download Price Variations via AJAX (this function can only be used
257  * in WordPress Admin). This function isused for the Edit Payment screen when downloads
258  * are added to the purchase. When each download is chosen, an AJAX call is fired
259  * to this function which will check if variable prices exist for that download.
260  * If they do, it will output a dropdown of all the variable prices available for
261  * that download.
262  *
263  * @author Sunny Ratilal
264  * @since 1.5
265  * @return void
266  */
267 function edd_check_for_download_price_variations() {
268     if ( ! check_ajax_referer( 'edd_add_downloads_to_purchase_nonce', 'nonce' ) )
269         return false;
270 
271     $download_id = intval( $_POST['download_id'] );
272 
273     if ( edd_has_variable_prices( $download_id ) ) {
274         $variable_prices = get_post_meta( $download_id, 'edd_variable_prices', true );
275 
276         if ( $variable_prices ) {
277             $ajax_response = '<select name="downloads[' . intval( $_POST['array_key'] ) . '][options][price_id]" class="edd-variable-prices-select">';
278                 foreach ( $variable_prices as $key => $price ) {
279                     $ajax_response .= '<option value="' . $key . '">' . $price['name']  . '</option>';
280                 }
281             $ajax_response .= '</select>';
282         }
283 
284         echo $ajax_response;
285     }
286 
287     edd_die();
288 }
289 add_action( 'wp_ajax_edd_check_for_download_price_variations', 'edd_check_for_download_price_variations' );

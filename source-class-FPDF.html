 575:             $w = $this->w - $this->rMargin - $this->x;

 577:         $s    = str_replace( "\r", '', $txt );
 578:         $nb   = strlen( $s );
 579:         if ( $nb > 0 && $s[$nb - 1] == "\n" )
 580:             $nb--;
 581:         $b = 0;
 582:         if ( $border ) {
 583:             if ( $border == 1 ) {
 584:                 $border = 'LTRB';
 585:                 $b      = 'LRT';
 586:                 $b2     = 'LR';
 587:             } else {
 588:                 $b2 = '';
 589:                 if ( strpos( $border, 'L' ) !== false )
 590:                     $b2 .= 'L';
 591:                 if ( strpos( $border, 'R' ) !== false )
 592:                     $b2 .= 'R';
 593:                 $b = ( strpos( $border, 'T' ) !== false ) ? $b2 . 'T' : $b2;
 594:             }
 595:         }
 596:         $sep = -1;
 597:         $i   = 0;
 598:         $j   = 0;
 599:         $l   = 0;
 600:         $ns  = 0;
 601:         $nl  = 1;
 602:         while ( $i < $nb ) {
 603:             $c = $s[$i];
 604:             if ( $c == "\n" ) {
 605:                 if ( $this->ws > 0 ) {
 606:                     $this->ws = 0;
 607:                     $this->_out( '0 Tw' );
 608:                 }
 609:                 $this->Cell( $w, $h, substr( $s, $j, $i - $j ), $b, 2, $align, $fill );
 610:                 $i++;

 612:                 $j   = $i;
 613:                 $l   = 0;
 614:                 $ns  = 0;
 615:                 $nl++;
 616:                 if ( $border && $nl == 2 )
 617:                     $b = $b2;
 618:                 continue;
 619:             }
 620:             if ( $c == ' ' ) {
 621:                 $sep = $i;
 622:                 $ls  = $l;
 623:                 $ns++;
 624:             }
 625:             $l += $cw[$c];
 626:             if ( $l > $wmax ) {
 627:                 if ( $sep == -1 ) {

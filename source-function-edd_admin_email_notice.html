 15 /**
 16  * Email the download link(s) and payment confirmation to the buyer in a
 17  * customizable Purchase Receipt
 18  *
 19  * @since 1.0
 20  * @param int $payment_id Payment ID
 21  * @param bool $admin_notice Whether to send the admin email notification or not (default: true)
 22  * @return void
 23  */
 24 function edd_email_purchase_receipt( $payment_id, $admin_notice = true ) {
 25     global $edd_options;
 26 
 27     $payment_data = edd_get_payment_meta( $payment_id );
 28     $user_info    = maybe_unserialize( $payment_data['user_info'] );
 29     $email        = edd_get_payment_user_email( $payment_id );
 30 
 31     if ( isset( $user_info['id'] ) && $user_info['id'] > 0 ) {
 32         $user_data = get_userdata($user_info['id']);
 33         $name = $user_data->display_name;
 34     } elseif ( isset( $user_info['first_name'] ) && isset( $user_info['last_name'] ) ) {
 35         $name = $user_info['first_name'] . ' ' . $user_info['last_name'];
 36     } else {
 37         $name = $email;
 38     }
 39 
 40     $message = edd_get_email_body_header();
 41     $message .= edd_get_email_body_content( $payment_id, $payment_data );
 42     $message .= edd_get_email_body_footer();
 43 
 44     $from_name = isset( $edd_options['from_name'] ) ? $edd_options['from_name'] : get_bloginfo('name');
 45     $from_email = isset( $edd_options['from_email'] ) ? $edd_options['from_email'] : get_option('admin_email');
 46 
 47     $subject = apply_filters( 'edd_purchase_subject', isset( $edd_options['purchase_subject'] )
 48         ? trim( $edd_options['purchase_subject'] )
 49         : __( 'Purchase Receipt', 'edd' ), $payment_id );
 50 
 51     $subject = edd_email_template_tags( $subject, $payment_data, $payment_id );
 52 
 53     $headers = "From: " . stripslashes_deep( html_entity_decode( $from_name, ENT_COMPAT, 'UTF-8' ) ) . " <$from_email>\r\n";
 54     $headers .= "Reply-To: ". $from_email . "\r\n";
 55     $headers .= "MIME-Version: 1.0\r\n";
 56     $headers .= "Content-Type: text/html; charset=utf-8\r\n";
 57 
 58     // Allow add-ons to add file attachments
 59     $attachments = apply_filters( 'edd_receipt_attachments', array(), $payment_id, $payment_data );
 60 
 61     wp_mail( $email, $subject, $message, $headers, $attachments );
 62 
 63     if ( $admin_notice && ! edd_admin_notices_disabled() ) {
 64         do_action( 'edd_admin_sale_notice', $payment_id, $payment_data );
 65     }
 66 }

112 function edd_admin_email_notice( $payment_id = 0, $payment_data = array() ) {
113 
114     /* Send an email notification to the admin */
115     $admin_email   = edd_get_admin_notice_emails();
116 
117     $user_info = maybe_unserialize( $payment_data['user_info'] );
118 
119     if ( isset( $user_info['id'] ) && $user_info['id'] > 0 ) {
120         $user_data = get_userdata($user_info['id']);
121         $name = $user_data->display_name;
122     } elseif ( isset( $user_info['first_name'] ) && isset($user_info['last_name'] ) ) {
123         $name = $user_info['first_name'] . ' ' . $user_info['last_name'];
124     } else {
125         $name = $user_info['email'];
126     }
127 
128     $admin_subject = apply_filters( 'edd_admin_purchase_notification_subject', __( 'New download purchase', 'edd' ), $payment_id, $payment_data );
129 
130     $admin_message = __( 'Hello', 'edd' ) . "\n\n" . sprintf( __( 'A %s purchase has been made', 'edd' ), edd_get_label_plural() ) . ".\n\n";
131     $admin_message .= sprintf( __( '%s sold:', 'edd' ), edd_get_label_plural() ) .  "\n\n";
132 
133     $download_list = '';
134     $downloads = maybe_unserialize( $payment_data['downloads'] );
135 
136     if ( is_array( $downloads ) ) {
137         foreach ( $downloads as $download ) {
138             $id    = isset( $payment_data['cart_details'] ) ? $download['id'] : $download;
139             $title = get_the_title( $id );
140             if ( isset( $download['options'] ) ) {
141                 if ( isset( $download['options']['price_id'] ) ) {
142                     $title .= ' - ' . edd_get_price_option_name( $id, $download['options']['price_id'], $payment_id );
143                 }
144             }
145             $download_list .= html_entity_decode( $title, ENT_COMPAT, 'UTF-8' ) . "\n";
146         }
147     }
148 
149     $gateway = edd_get_gateway_admin_label( get_post_meta( $payment_id, '_edd_payment_gateway', true ) );
150 
151     $admin_message .= $download_list . "\n";
152     $admin_message .= __( 'Purchased by: ', 'edd' )   . " " . html_entity_decode( $name, ENT_COMPAT, 'UTF-8' ) . "\n";
153     $admin_message .= __( 'Amount: ', 'edd' )         . " " . html_entity_decode( edd_currency_filter( edd_format_amount( $payment_data['amount'] ) ), ENT_COMPAT, 'UTF-8' ) . "\n\n";
154     $admin_message .= __( 'Payment Method: ', 'edd' ) . " " . $gateway . "\n\n";
155     $admin_message .= __( 'Thank you', 'edd' );
156     $admin_message = apply_filters( 'edd_admin_purchase_notification', $admin_message, $payment_id, $payment_data );
157 
158     $admin_headers = apply_filters( 'edd_admin_purchase_notification_headers', array(), $payment_id, $payment_data );
159 
160     $admin_attachments = apply_filters( 'edd_admin_purchase_notification_attachments', array(), $payment_id, $payment_data );
161 
162     wp_mail( $admin_email, $admin_subject, $admin_message, $admin_headers, $admin_attachments );
163 }

166 /**
167  * Retrieves the emails for which admin notifications are sent to (these can be
168  * changed in the EDD Settings)
169  *
170  * @since 1.0
171  * @global $edd_options Array of all the EDD Options
172  * @return void
173  */
174 function edd_get_admin_notice_emails() {
175     global $edd_options;
176 
177     $emails = isset( $edd_options['admin_notice_emails'] ) && strlen( trim( $edd_options['admin_notice_emails'] ) ) > 0 ? $edd_options['admin_notice_emails'] : get_bloginfo( 'admin_email' );
178     $emails = array_map( 'trim', explode( "\n", $emails ) );
179 
180     return apply_filters( 'edd_admin_notice_emails', $emails );
181 }

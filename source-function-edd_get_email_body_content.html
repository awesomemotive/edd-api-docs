 31 /**
 32  * Email Template Tags
 33  *
 34  * @since 1.0
 35  *
 36  * @param string $message Message with the template tags
 37  * @param array $payment_data Payment Data
 38  * @param int $payment_id Payment ID
 39  *
 40  * @return string $message Fully formatted message
 41  */
 42 function edd_email_template_tags( $message, $payment_data, $payment_id ) {
 43     global $edd_options;
 44 
 45     $has_tags = ( strpos($message, '{' ) !== false );
 46     if ( ! $has_tags ) return $message;
 47 
 48     $user_info = maybe_unserialize( $payment_data['user_info'] );
 49 
 50     $fullname = '';
 51     if ( isset( $user_info['id'] ) && $user_info['id'] > 0 && isset( $user_info['first_name'] ) ) {
 52         $user_data = get_userdata( $user_info['id'] );
 53         $name      = $user_info['first_name'];
 54         $fullname  = $user_info['first_name'] . ' ' . $user_info['last_name'];
 55         $username  = $user_data->user_login;
 56     } elseif ( isset( $user_info['first_name'] ) ) {
 57         $name     = $user_info['first_name'];
 58         $fullname = $user_info['first_name'] . ' ' . $user_info['last_name'];
 59         $username = $user_info['first_name'];
 60     } else {
 61         $name     = $user_info['email'];
 62         $username = $user_info['email'];
 63     }
 64 
 65     $file_urls     = '';
 66     $download_list = '<ul>';
 67     $cart_items     = edd_get_payment_meta_cart_details( $payment_id );
 68 
 69     if ( $cart_items ) {
 70         $show_names = apply_filters( 'edd_email_show_names', true );
 71 
 72         foreach ( $cart_items as $item ) {
 73 
 74             $price_id = edd_get_cart_item_price_id( $item );
 75 
 76             if ( $show_names ) {
 77 
 78                 $title = get_the_title( $item['id'] );
 79 
 80                 if( $price_id !== false )
 81                     $title .= "&nbsp;&ndash;&nbsp;" . edd_get_price_option_name( $item['id'], $price_id );
 82 
 83                 $download_list .= '<li>' . apply_filters( 'edd_email_receipt_download_title', $title, $item['id'], $price_id ) . '<br/>';
 84                 $download_list .= '<ul>';
 85             }
 86 
 87 
 88             $files = edd_get_download_files( $item['id'], $price_id );
 89 
 90             if ( $files ) {
 91                 foreach ( $files as $filekey => $file ) {
 92                     $download_list .= '<li>';
 93                     $file_url = edd_get_download_file_url( $payment_data['key'], $payment_data['email'], $filekey, $item['id'], $price_id );
 94                     $download_list .= '<a href="' . esc_url( $file_url ) . '">' . $file['name'] . '</a>';
 95 
 96                     $download_list .= '</li>';
 97 
 98                     $file_urls .= esc_html( $file_url ) . '<br/>';
 99                 }
100             }
101 
102             if ( $show_names ) {
103                 $download_list .= '</ul>';
104             }
105 
106             if ( '' != edd_get_product_notes( $item['id'] ) )
107                 $download_list .= ' &mdash; <small>' . edd_get_product_notes( $item['id'] ) . '</small>';
108 
109             if ( $show_names ) {
110                 $download_list .= '</li>';
111             }
112         }
113     }
114     $download_list .= '</ul>';
115 
116     $subtotal   = isset( $payment_data['subtotal'] ) ? $payment_data['subtotal'] : $payment_data['amount'];
117     $subtotal   = edd_currency_filter( edd_format_amount( $subtotal ) );
118     $tax        = isset( $payment_data['tax'] ) ? $payment_data['tax'] : 0;
119     $tax        = edd_currency_filter( edd_format_amount( $tax ) );
120     $price      = edd_currency_filter( edd_format_amount( $payment_data['amount'] ) );
121     $gateway    = edd_get_gateway_checkout_label( get_post_meta( $payment_id, '_edd_payment_gateway', true ) );
122     $receipt_id = $payment_data['key'];
123 
124     $message = str_replace( '{name}', $name, $message );
125     $message = str_replace( '{fullname}', $fullname, $message );
126     $message = str_replace( '{username}', $username, $message );
127     $message = str_replace( '{download_list}', $download_list, $message );
128     $message = str_replace( '{file_urls}', $file_urls, $message );
129     $message = str_replace( '{date}', date_i18n( get_option( 'date_format' ), strtotime( $payment_data['date'] ) ), $message );
130     $message = str_replace( '{sitename}', get_bloginfo( 'name' ), $message );
131     $message = str_replace( '{subtotal}', $subtotal, $message );
132     $message = str_replace( '{tax}', $tax, $message );
133     $message = str_replace( '{price}', $price, $message );
134     $message = str_replace( '{payment_method}', $gateway, $message );
135     $message = str_replace( '{receipt_id}', $receipt_id, $message );
136     $message = str_replace( '{payment_id}', $payment_id, $message );
137     $message = str_replace( '{receipt_link}', sprintf( __( '%1$sView it in your browser.%2$s', 'edd' ), '<a href="' . add_query_arg( array ( 'purchase_key' => $receipt_id, 'edd_action' => 'view_receipt' ), home_url() ) . '">', '</a>' ), $message );
138 
139     $message = apply_filters( 'edd_email_template_tags', $message, $payment_data, $payment_id );
140 
141     return $message;
142 }

301 /**
302  * Applies the Chosen Email Template
303  *
304  * @since 1.0.8.2
305  * @param string $body The contents of the receipt email
306  * @param int $payment_id The ID of the payment we are sending a receipt for
307  * @param array $payment_data An array of meta information for the payment
308  * @return string $email Formatted email with the template applied
309  */
310 function edd_apply_email_template( $body, $payment_id, $payment_data ) {
311     global $edd_options;
312 
313     $template_name = isset( $edd_options['email_template'] ) ? $edd_options['email_template'] : 'default';
314     $template_name = apply_filters( 'edd_email_template', $template_name, $payment_id );
315 
316     if ( $template_name == 'none' ) {
317         if ( is_admin() )
318             $body = edd_email_preview_templage_tags( $body );
319 
320         return $body; // Return the plain email with no template
321     }
322 
323     ob_start();
324 
325     do_action( 'edd_email_template_' . $template_name );
326 
327     $template = ob_get_clean();
328 
329     if ( is_admin() )
330         $body = edd_email_preview_templage_tags( $body );
331 
332     $body = apply_filters( 'edd_purchase_receipt_' . $template_name, $body );
333 
334     $email = str_replace( '{email}', $body, $template );
335 
336     return $email;
337 }

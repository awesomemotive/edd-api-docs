 62 /**
 63  * Displays a formatted price for a download
 64  *
 65  * @since 1.0
 66  * @param int $download_id ID of the download price to show
 67  * @param bool $echo Whether to echo or return the results
 68  * @return void
 69  */
 70 function edd_price( $download_id, $echo = true ) {
 71     if ( edd_has_variable_prices( $download_id ) ) {
 72         $prices = edd_get_variable_prices( $download_id );
 73         // Return the lowest price
 74         $price_float = 0;
 75         foreach ($prices as $key => $value)
 76             if ( ( ( (float)$prices[ $key ]['amount'] ) < $price_float ) or ( $price_float == 0 ) )
 77                 $price_float = (float)$prices[ $key ]['amount'];
 78             $price = edd_sanitize_amount( $price_float );
 79     } else {
 80         $price = edd_get_download_price( $download_id );
 81     }
 82 
 83     $price = apply_filters( 'edd_download_price', $price, $download_id );
 84 
 85     $price = '<span class="edd_price" id="edd_price_' . $download_id . '">' . $price . '</span>';
 86 
 87     if ( $echo )
 88         echo $price;
 89     else
 90         return $price;
 91 }

 95 /**
 96  * Retrieves the final price of a downloadable product after purchase
 97  * this price includes any necessary discounts that were applied
 98  *
 99  * @since 1.0
100  * @param int $download_id ID of the download
101  * @param array $user_purchase_info - an array of all information for the payment
102  * @param string $amount_override a custom amount that over rides the 'edd_price' meta, used for variable prices
103  * @return string - the price of the download
104  */
105 function edd_get_download_final_price( $download_id, $user_purchase_info, $amount_override = null ) {
106     if ( is_null( $amount_override ) ) {
107         $original_price = get_post_meta( $download_id, 'edd_price', true );
108     } else {
109         $original_price = $amount_override;
110     }
111     if ( isset( $user_purchase_info['discount'] ) && $user_purchase_info['discount'] != 'none' ) {
112         // if the discount was a %, we modify the amount. Flat rate discounts are ignored
113         if ( edd_get_discount_type( edd_get_discount_id_by_code( $user_purchase_info['discount'] ) ) != 'flat' )
114             $price = edd_get_discounted_amount( $user_purchase_info['discount'], $original_price );
115         else
116             $price = $original_price;
117     } else {
118         $price = $original_price;
119     }
120     return apply_filters( 'edd_final_price', $price, $download_id, $user_purchase_info );
121 }

170 /**
171  * Retrieves cheapest price option of a variable priced download
172  *
173  * @since 1.4.4
174  * @param int $download_id ID of the download
175  * @return float Amount of the lowest price
176  */
177 function edd_get_lowest_price_option( $download_id = 0 ) {
178     if ( empty( $download_id ) )
179         $download_id = get_the_ID();
180 
181     if ( ! edd_has_variable_prices( $download_id ) )
182         return edd_get_download_price( $download_id );
183 
184     $prices = edd_get_variable_prices( $download_id );
185 
186     $low = 0.00;
187 
188     if ( ! empty( $prices ) ) {
189         $min = 0;
190 
191         foreach ( $prices as $key => $price ) {
192             if ( empty( $price['amount'] ) )
193                 continue;
194             if ( $prices[ $min ]['amount'] > $price['amount'] )
195                 $min = $key;
196         }
197 
198         $low = $prices[ $min ]['amount'];
199     }
200 
201     return $low;
202 }

205  * Retrieves most expensive price option of a variable priced download
206  *
207  * @since 1.4.4
208  * @param int $download_id ID of the download
209  * @return float Amount of the highest price
210  */
211 function edd_get_highest_price_option( $download_id = 0 ) {
212     if ( empty( $download_id ) )
213         $download_id = get_the_ID();
214 
215     if ( ! edd_has_variable_prices( $download_id ) )
216         return edd_get_download_price( $download_id );
217 
218     $prices = edd_get_variable_prices( $download_id );
219 
220     $high = 0.00;
221 
222     if ( ! empty( $prices ) ) {
223         $max = 0;
224 
225         foreach ( $prices as $key => $price ) {
226             if ( empty( $price['amount'] ) )
227                 continue;
228 
229             if ( $prices[ $max ]['amount'] < $price['amount'] )
230                 $max = $key;
231         }
232 
233         $high = $prices[ $max ]['amount'];
234     }
235 
236     return $high;
237 }

653     $download_limit = edd_get_file_download_limit( $download_id );
654

730 /**
731  * Verifies a download purchase using a purchase key and email.
732  *
733  * @since 1.0
734  * @param int $download_id Download ID
735  * @param string $key
736  * @param string $email
737  * @param string $expire
738  * @param string $file_key
739  * @return bool True if payment and link was verified, false otherwise
740  */
741 function edd_verify_download_link( $download_id, $key, $email, $expire, $file_key ) {
742     $meta_query = array(
743         'relation'  => 'AND',
744         array(
745             'key'   => '_edd_payment_purchase_key',
746             'value' => $key
747         ),
748         array(
749             'key'   => '_edd_payment_user_email',
750             'value' => $email
751         )
752     );
753 
754     $payments = get_posts( array( 'meta_query' => $meta_query, 'post_type' => 'edd_payment' ) );
755 
756     if ( $payments ) {
757         foreach ( $payments as $payment ) {
758             $payment_meta   = edd_get_payment_meta( $payment->ID );
759             $downloads      = maybe_unserialize( $payment_meta['downloads'] );
760             $cart_details   = unserialize( $payment_meta['cart_details'] );
761 
762             if ( $payment->post_status != 'publish' && $payment->post_status != 'complete' )
763                 return false;
764 
765             if ( $downloads ) {
766                 foreach ( $downloads as $download_key => $download ) {
767 
768                     $id = isset( $payment_meta['cart_details'] ) ? $download['id'] : $download;
769 
770                     if ( $id != $download_id )
771                         continue;
772 
773                     $price_options = isset( $cart_details[ $download_key ]['item_number']['options'] ) ? $cart_details[ $download_key ]['item_number']['options'] : false;
774 
775                     $file_condition = edd_get_file_price_condition( $id, $file_key );
776 
777                     // If this download has variable prices, we have to confirm that this file was included in their purchase
778                     if ( ! empty( $price_options ) && $file_condition != 'all' && edd_has_variable_prices( $id ) ) {
779                         if ( $file_condition == $price_options['price_id'] )
780                             return $payment->ID;
781                     }
782 
783                     // Check to see if the file download limit has been reached
784                     if ( edd_is_file_at_download_limit( $id, $payment->ID, $file_key ) )
785                         wp_die( apply_filters( 'edd_download_limit_reached_text', __( 'Sorry but you have hit your download limit for this file.', 'edd' ) ), __( 'Error', 'edd' ) );
786 
787                     // Make sure the link hasn't expired
788                     if ( time() > $expire ) {
789                         wp_die( apply_filters( 'edd_download_link_expired_text', __( 'Sorry but your download link has expired.', 'edd' ) ), __( 'Error', 'edd' ) );
790                     }
791                     return $payment->ID; // Payment has been verified and link is still valid
792                 }
793 
794             }
795 
796         }
797 
798     }
799     // Payment not verified
800     return false;
801 }

311 /**
312  * Retrieves estimated monthly earnings and sales
313  *
314  * @since 1.5
315  * @return array
316  */
317 function edd_estimated_monthly_stats() {
318     $estimated = get_transient( 'edd_estimated_monthly_stats' );
319 
320     if ( false === $estimated ) {
321         $estimated = array(
322             'earnings' => 0,
323             'sales'    => 0
324         );
325 
326         $products = get_posts( array( 'post_type' => 'download', 'posts_per_page' => -1, 'fields' => 'ids' ) );
327         if ( $products ) {
328             foreach ( $products as $download ) {
329                 $estimated['earnings'] += edd_get_average_monthly_download_earnings( $download );
330                 $estimated['sales']    += number_format( edd_get_average_monthly_download_sales( $download ), 0 );
331             }
332         }
333 
334         // Cache for one day
335         set_transient( 'edd_estimated_monthly_stats', serialize( $estimated ), 86400 );
336     }
337 
338     return maybe_unserialize( $estimated );
339 }

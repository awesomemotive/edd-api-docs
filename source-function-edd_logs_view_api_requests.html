122 function edd_log_default_views() {
123     $views = array(

125         'sales'           => __( 'Sales', 'edd' ),
126         'gateway_errors'  => __( 'Payment Errors', 'edd' ),
127         'api_requests'    => __( 'API Requests', 'edd' )
128     );
129 
130     $views = apply_filters( 'edd_log_views', $views );
131 
132     return $views;
133 }

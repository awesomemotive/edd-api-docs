 78 /**
 79  * Register Styles
 80  *
 81  * Checks the styles option and hooks the required filter.
 82  *
 83  * @since 1.0
 84  * @global $edd_options
 85  * @return void
 86  */
 87 function edd_register_styles() {
 88     global $edd_options;
 89 
 90     if ( isset( $edd_options['disable_styles'] ) )
 91         return;
 92 
 93     $file = 'edd.css';
 94 
 95     if ( file_exists( trailingslashit( get_stylesheet_directory() ) . 'edd_templates/' . $file ) ) {
 96         $url = trailingslashit( get_stylesheet_directory_uri() ) . 'edd_templates/' . $file;
 97     } elseif ( file_exists( trailingslashit( get_template_directory() ) . 'edd_templates/' . $file ) ) {
 98         $url = trailingslashit( get_template_directory_uri() ) . 'edd_templates/' . $file;
 99     } elseif ( file_exists( trailingslashit( edd_get_templates_dir() ) . $file ) ) {
100         $url = trailingslashit( edd_get_templates_url() ) . $file;
101     }
102 
103     wp_enqueue_style( 'edd-styles', $url, EDD_VERSION );
104 }
105 add_action( 'wp_enqueue_scripts', 'edd_register_styles' );

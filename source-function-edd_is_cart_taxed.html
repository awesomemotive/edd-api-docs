108 /**

115 function edd_get_tax_rate() {
116     global $edd_options;
117 
118     $rate = isset( $edd_options['tax_rate'] ) ? (float) $edd_options['tax_rate'] : 0;
119 
120     if( $rate > 1 ) {

127 /**
128  * Calculate the taxed amount
129  *
130  * @since 1.3.3
131  * @param $amount float The original amount to calculate a tax cost
132  * @return float $tax Taxed amount
133  */
134 function edd_calculate_tax( $amount, $sum = true ) {
135     global $edd_options;
136 
137     // Not using taxes
138     if ( ! edd_use_taxes() ) return $amount;
139 
140     $rate = edd_get_tax_rate();
141     $tax = 0.00;
142     $prices_include_tax = edd_prices_include_tax();
143 
144     if ( $prices_include_tax ) {
145         $tax = $amount - ( $amount / ( $rate + 1 ) );
146     } else {
147         $tax = $amount * $rate;
148     }
149 
150     if ( $sum ) {
151 
152         if ( $prices_include_tax ) {
153             $tax = $amount - $tax;
154         } else {
155             $tax = $amount + $tax;
156         }
157 
158     }
159     return apply_filters( 'edd_taxed_amount', round( $tax, 2 ), $rate );
160 }

174 /**
175  * Gets the sales tax for the current year
176  *
177  * @since 1.3.3
178  * @param $year int The year to retrieve taxes for, i.e. 2012
179  * @uses edd_get_payment_tax()
180  * @return float $tax Sales tax
181  */
182 function edd_get_sales_tax_for_year( $year = null ) {
183     if ( empty( $year ) )
184         return 0;
185 
186     // Start at zero
187     $tax = 0;
188 
189     $args = array(
190         'post_type'         => 'edd_payment',
191         'posts_per_page'    => -1,
192         'year'              => $year,
193         'meta_key'          => '_edd_payment_mode',
194         'meta_value'        => edd_is_test_mode() ? 'test' : 'live',
195         'fields'            => 'ids'
196     );
197 
198     $payments = get_posts( $args );
199 
200     if( $payments ) :
201 
202         foreach( $payments as $payment ) :
203             $tax += edd_get_payment_tax( $payment );
204         endforeach;
205 
206     endif;
207 
208     return apply_filters( 'edd_get_sales_tax_for_year', $tax, $year );
209 }
